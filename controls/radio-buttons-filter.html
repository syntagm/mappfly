HTML Layout:

Example 

						
<!-- radio buttons filters -->
<div class="radio-filters">
	
	<div class="radio">
		<input 
			data-control-type="radio-buttons-filters"
			data-control-action="filter"
			data-control-name="default" 
			data-path="default" 
			
			id="default-radio" 
			type="radio" 
			name="jplist"
			checked="checked"
		/> 
		
		<label for="default-radio">All</label>
	</div>
	
	<div class="radio">
		<input 									
			data-control-type="radio-buttons-filters"

		/>
